    # 十个例子带你入门ssd1306屏幕(mircopython)

    # 介绍一下环境和硬件部分

    我的开发板是树莓派pico, 他的优点就是引脚多, 强大的2040处理器, 关键是便宜, 只需30元以内就可以买到。

    使用的屏幕是基于ssd1306的四引脚OLED屏幕, 我根据micropython官方教程 (opens new window)找到了这块屏幕的示例。

    接线图图示:

    # 例0: ssd1306.py (opens new window), 请将第三方库的代码保存到单片机中(没有这一步其他都白搭)

    # 例1: I2C初始化ssd1306

    from machine import SoftI2C,Pin
# 首先从machine库里面调出 I2C通讯模块 和 Pin引脚控制模块
from ssd1306 import SSD1306_I2C

'''自定义scl和sda对应的接线端口'''
i2c=SoftI2C(scl=Pin(4),sda=Pin(3)) 
#scl,sda在屏幕后方有标注, 接线根据个人情况调整
'''自定义屏幕的像素长度和宽度'''
display=SSD1306_I2C(128,64,i2c) 
#128是屏幕长度, 64是屏幕宽度, 单位:像素(px), 根据个人情况调整 
display.text("HealthIT Released",0,0) #(文本内容,起始x坐标, 起始y坐标)
display.show() #没有这一行, 屏幕不会显示任何内容

    # 例2: 点亮和熄灭一块屏幕

    from machine import SoftI2C,Pin
from ssd1306 import SSD1306_I2C
import time
i2c=SoftI2C(scl=Pin(4),sda=Pin(3)) 
display=SSD1306_I2C(128,64,i2c)

display.fill(1) #整片屏幕全亮
display.show() 
while True:
    display.poweroff() #关闭显示器电源，像素保留在内存中
    time.sleep(2)
    display.poweron() #打开显示器电源，重新绘制像素
    time.sleep(2)

"""通俗解释
display.poweroff()#关闭显示器电源，像素保留在内存中
display.poweron()#打开显示器电源，重新绘制像素
# 经过测试, 这样的方法打开/关闭显示器, 像素内容无需重新设置, 均保存在内存中, 再次show即可展示。
# 通俗的讲, 就是一个开关按键, 只是暂停显示, 因此不用再次display一次, 屏幕中显示的内容不会清空
"""

    # 例3: 调整屏幕的亮度

    from machine import SoftI2C,Pin
from ssd1306 import SSD1306_I2C
import time
i2c=SoftI2C(scl=Pin(4),sda=Pin(3)) 
display=SSD1306_I2C(128,64,i2c)

display.fill(1) #整片屏幕全亮
display.show() 
for i in range(0,255):
    print("亮度:",i)
    display.contrast(i) #关闭显示器电源，像素保留在内存中
    time.sleep(0.01)

"""解释
display.contrast(0) #dim,此时亮度最暗
display.contrast(255)#bright, 此时亮度最亮
# 经过测试, 这相当于一个亮度调节器, 你可以随时调整显示的亮度(同样的, 不需要display.show刷新图像)
"""

    # 例6: 由display, 顺便科普一下FrameBuffer

    # FrameBuffer科普

    Framebuffer，也叫帧缓冲，其内容对应于屏幕上的界面显示，可以将其简单理解为屏幕上显示内容对应的缓存，修改Framebuffer中的内容，即表示修改屏幕上的内容，所以，直接操作Framebuffer可以直接从显示器上观察到效果。

    • 说的更直白一些, framebuffer就是一个临时存放数据的区域, 对应OLED或LCD屏幕的显存(或内存)
    • 如果你想了解更多FrameBuffer的内容, 看这个链接 (opens new window)

    # 例9: 画个矩形框

    from machine import SoftI2C,Pin
from ssd1306 import SSD1306_I2C
import time
import random
i2c=SoftI2C(scl=Pin(4),sda=Pin(3)) 
display=SSD1306_I2C(128,64,i2c)

# 实验1: rect绘制矩形框
for i in range(0,30):
    # 起始坐标x1=0, y1=0; 结束坐标x2=循环数*4, y2=循环数*4, 颜色为1 ## 从左上角一直向下衍生的空心矩形
    display.rect(0, 0, i*4, i*4, 1)
    time.sleep(0.05)
    display.show()

# 实验2: fill_rect绘制实心矩形框
display.fill(0)
for i in range(0,30):
    display.fill(0)
    time.sleep(0.2)
    x=random.randint(10,128)
    y=random.randint(10,64)
    # display.rect(10, 10, 107, 43, 1)# 起始坐标x1=10, y1=10; 结束坐标x2=107, y2=43, 颜色为1
    # display.fill_rect(x-1, y-1, x, y, 1)
    display.fill_rect(x-10, y-10, x, y, 1)
    print(f"实心矩形始于:({x-10},{y-10}); 终于({x},{y})")
    time.sleep(0.2)
    display.show()

    # 例10: scoll实现元素的滚动~

    from machine import SoftI2C,Pin

import time
import random
i2c=SoftI2C(scl=Pin(4),sda=Pin(3))

display.show()
for i in range(0,100):
    if (i%10==0):
        display.fill(0)

        display.show()
    else:
        time.sleep(0.2)
        display.scroll(5, 5)
        display.show()
