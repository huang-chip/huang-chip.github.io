        # 如何使用百度菜品识别API识别菜品

        # 在阅读本篇文章之前, 你需要掌握Python基础

        # 01 注册/登录百度账号

        进入百度AI开放平台-菜品识别官网 (opens new window)

        点击该页面的 "控制台" 进行登陆操作

        (opens new window)

        # 02 进入官方文档查看代码

        官方帮助文档 > 图像识别 (opens new window)

        # 官方提供的python代码, 以此为例进行讲解
import requests
import base64
'''
菜品识别
'''

request_url = "https://aip.baidubce.com/rest/2.0/image-classify/v2/dish"
# 二进制方式打开图片文件
f = open('[本地文件]', 'rb')
img = base64.b64encode(f.read())

params = {"image":img,"top_num":5}
'''
------------------------------
params存储的是请求参数
请参照图1.1, 修改你需要的请求参数
------------------------------
'''
access_token = '[调用鉴权接口获取的token]'
'''
----------------------------------
access_token的获取方法请见代码块1.2
----------------------------------
'''
request_url = request_url + "?access_token=" + access_token
headers = {'content-type': 'application/x-www-form-urlencoded'}
response = requests.post(request_url, data=params, headers=headers)
if response:
    print (response.json())

        # 运行如下代码即可获取assess_token

        import requests
import json

url = "https://aip.baidubce.com/oauth/2.0/token"
data = {
    'grant_type':'client_credentials',
    'client_id':'[自己的API Key]',
    'client_secret':'[自己的Secret Key]'
}

response = requests.post(url=url,data=data)
print(response.text)

temp1 = json.loads(response.text)
print(f"\n\n 'access_token': {temp1['access_token']}")
