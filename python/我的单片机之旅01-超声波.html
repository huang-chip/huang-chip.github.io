    # 引脚分析

    VCC : 电源引脚
    Trig : 触发引脚
    Echo : 回声引脚
    GND : 地线引脚

    # 原理和技术流程描述

    1. 由于 一开始Trig引脚 和 Echo引脚是不带电的,
    2. 我们首先用单片机的P1口(仅作示例) 对Trig引脚给一个至少10ms(微秒)的高电平(5V)
    3. 然后使用单片机的P2口(仅作示例) 监测Echo引脚有没有变成高电平
    4. 当变成高电平后, 单片机立即开始计时, 直到Echo的电平变为0,停止计时, 这个时间可以用来计算距离
    • 340m/s == 340*10^-3mm/10^3ms == 340*10^-6mm/ms
    1. 计算公式: 时间*速度/2 == 距离

from machine import Pin,Timer
# Pin类用来控制引脚
# Timer类用来驱动一个硬件定时器
import utime
# utime类起到测量时间间隔的功能 
trigger = Pin(10,Pin.OUT)
# trigger引脚为板子连接Trig的口, 为输出口
echo = Pin(12,Pin.IN)
# echo引脚为板子连接Echo的口, 为输入口
tim = Timer()
# tim为一个定时器对象
def getdate(timer):
    # 告诉底层函数, 该函数多少秒执行一次
    trigger.low()
    utime.sleep_us(2)
    # 先让Trig处于低电平状态 2微秒
    trigger.high()
    utime.sleep_us(10)
    # 再让Trig处于低电平状态 10微秒
    trigger.low()
    # 调用结束后, 再给它一个低电平
    while echo.value() == 0:
        signalend = utime.ticks_us()
    # 当Echo引脚为0时, 运行计时器(开始计时)
    # 当Echo引脚为1时, 开始接收到波, signalend保存为(起始时间)
    while echo.value() == 1:
        signalon = utime.ticks_us()
    # 当Echo引脚为1时, 开始计时
    # 当Echo引脚为0时, 不再接收到波, signalon保存为(起始时间)
    timepassed = signalon-signalend
    # timepassed存储的是Echo自接收到信号到停止接受信号的(时间间隔)
    distance = (timepassed * 0.0343) / 2
    # 从米/秒-->厘米/微秒; 除以二就是到障碍物的距离
    print("The distance from object is ",distance,"cm")
